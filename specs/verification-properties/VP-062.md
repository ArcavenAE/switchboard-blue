# VP-062: JSON Output is Valid for All sbctl Metrics CLI Input Combinations

## Property Statement

∀ invocation of any sbctl metrics subcommand (`paths list`, `router metrics`,
`router status` alias) with `--json` flag and any combination of valid and
borderline flag arguments:

1. The command either exits 0 with output that parses as valid JSON, OR exits
   non-zero and emits a plain-text error message (not malformed JSON).
2. The JSON output, when exit code is 0, unmarshals without error using
   `encoding/json.Unmarshal`.
3. The `router status` alias produces output with the same top-level JSON schema
   as `paths list` (i.e. it is structurally identical except for the additional
   `quality` field per EC-005).
4. No input combination causes a panic, nil-dereference crash, or non-JSON bytes
   mixed into the JSON output stream.
5. **[v1.1 — F-T3-003]** When `rtt_p99_ms` is `"pending"` (i.e. `RTTValue{Valid: false}`),
   the `quality` field in the alias response MUST be `"pending"` (not
   green/yellow/red). The fuzz harness and proptest MUST assert this sentinel
   propagation for all inputs where `rttP99Valid=false` (BC-2.06.003 v1.5 PC-3 + EC-006).

This property enforces BC-2.06.003 Postcondition 4, EC-005 (alias schema
equivalence), and EC-006 (pending-p99 quality sentinel propagation, v1.1). It is
verified by fuzz testing the CLI dispatch layer and by property-based testing of
the JSON serialization routines directly.

**Verification phase:** Phase 6 (formal hardening). Not blocking Wave 5
implementation — correctness of individual command outputs is verified by VP-047
(integration). This property provides the well-formedness guarantee across all
input combinations including the alias form.

## Source Contract

- **BC:** BC-2.06.003 v1.5 — Per-path RTT and loss metrics queryable via sbctl
- **Postcondition 4:** "Metrics are returned as JSON with `--json` flag. Both the
  canonical form and the alias respect `--json`."
- **EC-005:** "Output is identical to `sbctl paths list` plus a `quality` column
  (green/yellow/red). Exit code, JSON schema, and error handling are identical."
- **EC-006 [v1.1]:** "`rtt_p99_ms` is `"pending"` AND `quality` is `"pending"` when
  fewer than 10 RTT samples collected. The quality column MUST NOT be green/yellow/red
  when p99 is pending."

## Proof Method

| Method | Tool | Bounded? | Coverage |
|--------|------|----------|----------|
| fuzz | go test -fuzz (stdlib) | no (time-bounded) | Feed arbitrary byte sequences as flag values and daemon response payloads; assert JSON well-formedness of every 0-exit response |
| proptest | gopter | no | For all (path list length N, rtt_p99 pending/value, quality indicator) combinations: assert JSON schema consistency between paths list and router status alias |

### Fuzz Corpus Seeds

The following inputs seed the fuzz corpus (stored in `testdata/fuzz/FuzzSbctlMetricsJSON/`):

1. `paths list --json` with empty paths response
2. `paths list --json` with 1 path, `rtt_p99_ms: "pending"`
3. `paths list --json` with 5 paths, all fields populated
4. `router metrics --svtn=abc123 --json` with full response
5. `router status --target 127.0.0.1:9000 --json` with alias response
6. `paths list --json` against unreachable daemon (E-NET-001 path)
7. **[v1.1]** `router status --target 127.0.0.1:9000 --json` with `rttP99Valid=false`
   (pending p99); assert `alias["quality"] == "pending"` (EC-006 sentinel)

## Proof Harness Skeleton

```go
package sbctl_test

import (
	"bytes"
	"encoding/json"
	"testing"

	"github.com/arcavenae/switchboard/cmd/sbctl/internal/metricshandler"
	"github.com/arcavenae/switchboard/internal/metrics"
)

// FuzzSbctlMetricsJSON verifies VP-062: all metrics serialization paths produce
// valid JSON regardless of the content of the metric fields.
//
// The fuzzer drives the internal serialization layer (not the subprocess CLI)
// to avoid subprocess overhead while maximizing coverage of the marshaling path.
func FuzzSbctlMetricsJSON(f *testing.F) {
	// Seed corpus: representative PathEntry field combinations.
	f.Add("path-001", "127.0.0.1:9000", 15.0, true, 22.0, 0.1, "active")
	f.Add("", "", 0.0, false, 0.0, 0.0, "")
	f.Add("path-999", "10.0.0.1:1", 9999.9, true, 9999.9, 100.0, "failed")
	// v1.1: pending-p99 seed — rttP99Valid=false triggers quality:"pending" (EC-006)
	f.Add("path-002", "10.0.0.2:9000", 12.0, false, 0.0, 0.0, "active")

	f.Fuzz(func(t *testing.T,
		pathID string,
		routerAddr string,
		rttMs float64,
		rttP99Valid bool,
		rttP99Value float64,
		lossPct float64,
		status string,
	) {
		entry := metrics.PathEntry{
			PathID:     pathID,
			RouterAddr: routerAddr,
			RTTMs:      rttMs,
			RTTp99Ms:   metrics.RTTValue{Valid: rttP99Valid, Value: rttP99Value},
			LossPct:    lossPct,
			Status:     status,
		}

		// Verify PathsListResponse serializes to valid JSON.
		resp := metrics.PathsListResponse{Paths: []metrics.PathEntry{entry}}
		data, err := json.Marshal(resp)
		if err != nil {
			t.Fatalf("json.Marshal(PathsListResponse): %v", err)
		}
		if !json.Valid(data) {
			t.Fatalf("PathsListResponse produced invalid JSON: %s", data)
		}

		// Verify RouterStatusResponse (alias form with quality field) is
		// also valid JSON and shares the paths array schema.
		aliasResp := metricshandler.RouterStatusResponse{
			PathsListResponse: resp,
			Quality:           metricshandler.DeriveQuality(entry),
		}
		aliasData, err := json.Marshal(aliasResp)
		if err != nil {
			t.Fatalf("json.Marshal(RouterStatusResponse): %v", err)
		}
		if !json.Valid(aliasData) {
			t.Fatalf("RouterStatusResponse produced invalid JSON: %s", aliasData)
		}

		// Verify alias output is a strict superset of paths list output
		// (same paths array, additional quality field only).
		var base, alias map[string]json.RawMessage
		if err := json.Unmarshal(data, &base); err != nil {
			t.Fatalf("unmarshal base: %v", err)
		}
		if err := json.Unmarshal(aliasData, &alias); err != nil {
			t.Fatalf("unmarshal alias: %v", err)
		}
		if !bytes.Equal(base["paths"], alias["paths"]) {
			t.Errorf("alias paths array differs from canonical paths array (EC-005 violation)")
		}
		if _, ok := alias["quality"]; !ok {
			t.Error("alias response missing quality field (EC-005 violation)")
		}
		// v1.1: EC-006 pending-quality sentinel assertion.
		// When rttP99Valid=false the quality field MUST be "pending" (BC-2.06.003 v1.5 PC-3).
		if !rttP99Valid {
			var qualVal string
			if err := json.Unmarshal(alias["quality"], &qualVal); err != nil {
				t.Fatalf("unmarshal quality field: %v", err)
			}
			if qualVal != "pending" {
				t.Errorf("EC-006 violation: quality=%q for pending p99; want \"pending\"", qualVal)
			}
		}
	})
}

// TestSbctlMetricsJSON_PropTest verifies VP-062 property 3 with explicit cases:
// router status alias schema = paths list schema + quality field.
func TestSbctlMetricsJSON_PropTest(t *testing.T) {
	t.Parallel()

	cases := []struct {
		name      string
		entries   []metrics.PathEntry
		wantQuals []string
	}{
		{
			name:      "single active path",
			entries:   []metrics.PathEntry{{PathID: "p1", RouterAddr: "127.0.0.1:9000", RTTMs: 15.0, LossPct: 0.0, Status: "active"}},
			wantQuals: []string{"green"},
		},
		{
			name:      "single degraded path",
			entries:   []metrics.PathEntry{{PathID: "p1", RouterAddr: "127.0.0.1:9000", RTTMs: 250.0, LossPct: 2.0, Status: "degraded"}},
			wantQuals: []string{"yellow"},
		},
		{
			name:      "single failed path",
			entries:   []metrics.PathEntry{{PathID: "p1", RouterAddr: "127.0.0.1:9000", RTTMs: 0.0, LossPct: 0.0, Status: "failed"}},
			wantQuals: []string{"red"},
		},
		{
			name:    "empty path list",
			entries: []metrics.PathEntry{},
		},
		// v1.1: EC-006 pending-quality sentinel (BC-2.06.003 v1.5 PC-3 + EC-006).
		{
			name:      "pending p99 path",
			entries:   []metrics.PathEntry{{PathID: "p1", RouterAddr: "127.0.0.1:9000", RTTMs: 12.0, RTTp99Ms: metrics.RTTValue{Valid: false}, LossPct: 0.0, Status: "active"}},
			wantQuals: []string{"pending"},
		},
	}

	for _, tc := range cases {
		tc := tc
		t.Run(tc.name, func(t *testing.T) {
			t.Parallel()

			resp := metrics.PathsListResponse{Paths: tc.entries}
			data, err := json.Marshal(resp)
			if err != nil {
				t.Fatalf("json.Marshal: %v", err)
			}
			if !json.Valid(data) {
				t.Fatalf("invalid JSON: %s", data)
			}
		})
	}
}
```
