# VP-063: PathTracker.IsDegraded() Reflects EWMA-RTT vs DegradedRTTThresholdMS

## Property Statement

For any RTT sample sequence fed into PathTracker, IsDegraded() is true if and
only if the EWMA-smoothed RTT exceeds DegradedRTTThresholdMS (200.0 ms).
Recovery below the threshold clears the degraded flag.

Formally:

```
∀ sample sequence [rtt₁, rtt₂, …, rttₙ]:
  let ewma_n = EWMA(rtt₁…rttₙ)
  IsDegraded() = (ewma_n > DegradedRTTThresholdMS)
```

The flag must track the EWMA value continuously — both onset (EWMA crosses
200 ms upward) and recovery (EWMA crosses 200 ms downward) must be reflected
in the returned boolean with no hysteresis or additional delay.

## Source Contract

- **BC:** BC-2.02.003 — Per-path RTT and loss tracked via keep-alive probes; paths ranked by quality
- **Postcondition:** PC-5 — "A path whose RTT exceeds the degradation threshold (implementation: >200ms) is flagged as degraded."

## Proof Harness Skeleton

```go
package paths_test

import (
	"testing"

	"github.com/leanovate/gopter"
	"github.com/leanovate/gopter/gen"
	"github.com/leanovate/gopter/prop"

	"github.com/arcavenae/switchboard/internal/paths"
)

const DegradedRTTThresholdMS = 200.0

// genRTTSamples generates a non-empty slice of RTT values in [0, 500] ms.
func genRTTSamples() gopter.Gen {
	return gen.SliceOf(gen.Float64Range(0, 500)).
		SuchThat(func(v interface{}) bool {
			return len(v.([]float64)) > 0
		})
}

func TestProp_IsDegraded_TracksEWMAThreshold(t *testing.T) {
	properties := gopter.NewProperties(nil)

	properties.Property("IsDegraded tracks EWMA vs DegradedRTTThresholdMS", prop.ForAll(
		func(samples []float64) bool {
			tracker := paths.NewPathTracker()
			var ewma float64
			const alpha = 0.2 // must match PathTracker's EWMA alpha

			for _, rtt := range samples {
				if ewma == 0 {
					ewma = rtt
				} else {
					ewma = alpha*rtt + (1-alpha)*ewma
				}
				tracker.RecordRTT(rtt)
			}

			expected := ewma > DegradedRTTThresholdMS
			return tracker.IsDegraded() == expected
		},
		genRTTSamples(),
	))

	properties.TestingRun(t)
}

func TestProp_IsDegraded_RecoveryClears(t *testing.T) {
	properties := gopter.NewProperties(nil)

	// Property: after a high-RTT phase followed by sustained low-RTT probes,
	// IsDegraded() eventually becomes false.
	properties.Property("IsDegraded clears after recovery sequence", prop.ForAll(
		func(highRTT float64, recoveryCount uint8) bool {
			if recoveryCount == 0 {
				recoveryCount = 1
			}
			tracker := paths.NewPathTracker()

			// Drive EWMA above threshold.
			for i := 0; i < 20; i++ {
				tracker.RecordRTT(highRTT + 100.0)
			}

			// Feed low-RTT recovery samples well below threshold.
			lowRTT := 10.0
			for i := 0; i < int(recoveryCount)+20; i++ {
				tracker.RecordRTT(lowRTT)
			}

			return !tracker.IsDegraded()
		},
		gen.Float64Range(200.1, 400.0),
		gen.UInt8Range(1, 50),
	))

	properties.TestingRun(t)
}
```
