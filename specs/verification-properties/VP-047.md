# VP-047: Per-Path Metrics Queryable via sbctl

## Property Statement

`sbctl paths list --json` returns a JSON response where:
  len(response.paths) ≥ 1
  ∧ ∀ path p in response.paths:
      p.path_id is present
      ∧ p.router_addr is present and equals `PathSnapshot.RouterAddr`;
          `""` only for paths constructed via addr-less `NewPathTracker`
          (end-to-end observability with non-empty value requires
          S-BL.PATH-TRACKER-WIRING)
      ∧ p.rtt_ms is present
      ∧ p.rtt_p99_ms is present (float64 or `"pending"` when < 10 samples; never absent)
      ∧ p.loss_pct is present
      ∧ p.status is present and is one of `{active, degraded}` (`failed` is reserved; MUST NOT appear in Wave-6 responses)

Per-path metrics must be machine-readable via the CLI. Operators and
monitoring systems depend on these fields to assess path health.

**Note (post-S-BL.ROUTER-ADDR):** `router_addr` MUST equal `PathSnapshot.RouterAddr`.
`""` is only valid for paths whose `PathTracker` was constructed via addr-less
`NewPathTracker`. Integration tests using a `PathsListSource` stub with
`NewPathTrackerWithAddr("127.0.0.1:9000", ...)` MUST assert `router_addr ==
"127.0.0.1:9000"`, not accept `""`. DRIFT-SW504-ROUTER_ADDR-PLACEHOLDER is closed
(BC-2.06.003 v1.15; S-BL.ROUTER-ADDR).

## Source Contract

- **BC:** BC-2.06.003 — Per-path RTT and loss metrics queryable via sbctl
- **Postcondition/Invariant:** Postcondition — "sbctl paths list --json returns paths with rtt_ms, loss_pct, and status."

## Proof Harness Skeleton

```go
//go:build integration

package metrics_test

import (
	"bytes"
	"context"
	"encoding/json"
	"os/exec"
	"testing"
	"time"

	"github.com/arcavenae/switchboard/internal/testenv"
)
// Note: encoding/json is used both for Unmarshal and for json.RawMessage (rtt_p99_ms field).

// NOTE (F-T4): The JSON envelope is defined in interface-definitions.md v1.3+ as:
//   {"ok": true, "data": {"paths": [...]}}
// The BC-2.06.003 Canonical Test Vectors (§Canonical Test Vectors) show bare
// path arrays; those vectors document the per-path entry schema only.
// interface-definitions.md v1.3 is the authority for the outer envelope shape.
// The structs below reflect the full enveloped response.

type pathEntry struct {
	PathID     *string      `json:"path_id"`
	// RouterAddr: MUST equal PathSnapshot.RouterAddr. Non-empty when PathTracker
	// constructed via NewPathTrackerWithAddr. Empty string only for addr-less
	// NewPathTracker-constructed paths. DRIFT-SW504-ROUTER_ADDR-PLACEHOLDER closed.
	RouterAddr *string      `json:"router_addr"`
	RTTMs      *float64     `json:"rtt_ms"`
	// RTTp99Ms: float64 when ≥10 samples; the string "pending" when < 10 samples.
	// Represented as json.RawMessage to handle both number and string variants.
	RTTp99Ms   json.RawMessage `json:"rtt_p99_ms"`
	LossPct    *float64     `json:"loss_pct"`
	Status     *string      `json:"status"`
}

type pathsData struct {
	Paths []pathEntry `json:"paths"`
}

type pathsEnvelope struct {
	OK   bool       `json:"ok"`
	Data *pathsData `json:"data"`
}

// TestIntegration_PerPathMetrics_QueryableViaSbctl verifies that
// `sbctl paths list --json` returns at least one path with required fields.
func TestIntegration_PerPathMetrics_QueryableViaSbctl(t *testing.T) {
	t.Parallel()

	ctx, cancel := context.WithTimeout(context.Background(), 15*time.Second)
	t.Cleanup(cancel)

	env := testenv.New(t, ctx)
	t.Cleanup(env.Close)

	env.CreateSession(t)

	// Allow metrics to accumulate.
	time.Sleep(500 * time.Millisecond)

	cmd := exec.CommandContext(ctx, "sbctl",
		"--target", env.RouterAddr(t),
		"paths", "list", "--json",
	)
	var stdout, stderr bytes.Buffer
	cmd.Stdout = &stdout
	cmd.Stderr = &stderr

	if err := cmd.Run(); err != nil {
		t.Fatalf("sbctl paths list --json failed: %v\nstderr: %s", err, stderr.String())
	}

	var env2 pathsEnvelope
	if err := json.Unmarshal(stdout.Bytes(), &env2); err != nil {
		t.Fatalf("failed to parse JSON envelope: %v\nbody: %s", err, stdout.String())
	}
	if !env2.OK {
		t.Fatalf("envelope ok=false; body: %s", stdout.String())
	}
	if env2.Data == nil {
		t.Fatal("envelope data is null")
	}

	if len(env2.Data.Paths) == 0 {
		t.Fatal("expected at least one path in response; got empty list")
	}

	for i, p := range env2.Data.Paths {
		if p.PathID == nil {
			t.Errorf("path[%d]: missing path_id field", i)
		}
		// router_addr must equal PathSnapshot.RouterAddr. Non-empty for paths constructed
		// via NewPathTrackerWithAddr. Assert key presence; value check depends on stub.
		if p.RouterAddr == nil {
			t.Errorf("path[%d]: missing router_addr field", i)
		}
		if p.RTTMs == nil {
			t.Errorf("path[%d]: missing rtt_ms field", i)
		}
		if len(p.RTTp99Ms) == 0 {
			t.Errorf("path[%d]: missing rtt_p99_ms field (must be float64 or \"pending\")", i)
		}
		if p.LossPct == nil {
			t.Errorf("path[%d]: missing loss_pct field", i)
		}
		if p.Status == nil || *p.Status == "" {
			t.Errorf("path[%d]: missing or empty status field", i)
		}
		// Ruling-4: status must be "active" or "degraded" in Wave 6; "failed" is reserved.
		if p.Status != nil && *p.Status == "failed" {
			t.Errorf("path[%d]: status=\"failed\" is reserved and MUST NOT appear in Wave-6 responses", i)
		}
	}
}
```
