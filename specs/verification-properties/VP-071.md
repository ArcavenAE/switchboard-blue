# VP-071: Handler Execution Error Returns E-RPC-011 In-Band; Connection Not Closed

## Property Statement

When a registered handler's `Fn` returns a non-nil error, `mgmt.Server` MUST:

1. Send an in-band error response:
   ```json
   {"type":"response","id":"<id>","ok":false,"error":{"code":"E-RPC-011","message":"<err>"},"data":null}
   ```
   where `<id>` matches the request's `id` field and `<err>` is the handler's
   error string verbatim (not further wrapped by `internal/mgmt`).
2. NOT close the connection. The client MUST be able to send further RPC requests
   on the same connection after receiving this response.
3. NOT surface `E-RPC-002`. This code must not appear in `internal/mgmt`.

This is server-side only and distinct from `E-RPC-001` (client) and `E-RPC-010`
(unknown command).

## Source Contract

- **BC:** BC-2.07.004 PC-12 (Ruling C) — "handler returns error → in-band response
  ok:false code E-RPC-011 '<err>', connection NOT closed."

```go
package mgmt_test

import (
    "context"
    "crypto/ed25519"
    "crypto/rand"
    "encoding/json"
    "errors"
    "net"
    "testing"
    "time"

    "github.com/arcavenae/switchboard/internal/mgmt"
)

// TestHandlerError_InBandError_VP071 verifies VP-071:
// Handler error → E-RPC-011 in-band; connection remains open.
func TestHandlerError_InBandError_VP071(t *testing.T) {
    t.Parallel()

    ln, err := net.Listen("tcp", "127.0.0.1:0")
    if err != nil {
        t.Fatalf("listen: %v", err)
    }
    t.Cleanup(func() { ln.Close() })

    _, daemonKey, _ := ed25519.GenerateKey(rand.Reader)
    opPub, opPriv, _ := ed25519.GenerateKey(rand.Reader)
    keySet := mgmt.NewOperatorKeySet([]ed25519.PublicKey{opPub})

    const handlerErrMsg = "simulated handler failure"
    handlers := []mgmt.Handler{
        {
            Command: "test.fail",
            Fn: func(ctx context.Context, args json.RawMessage) (interface{}, error) {
                return nil, errors.New(handlerErrMsg)
            },
        },
        {
            Command: "test.ok",
            Fn: func(ctx context.Context, args json.RawMessage) (interface{}, error) {
                return map[string]string{"status": "ok"}, nil
            },
        },
    }

    srv := mgmt.NewServer(ln, daemonKey, keySet, handlers, "dev")
    ctx, cancel := context.WithCancel(context.Background())
    t.Cleanup(cancel)
    go srv.Serve(ctx) //nolint:errcheck

    conn, err := net.DialTimeout("tcp", ln.Addr().String(), time.Second)
    if err != nil {
        t.Fatalf("dial: %v", err)
    }
    t.Cleanup(func() { conn.Close() })

    if err := doHandshake(t, conn, opPriv); err != nil {
        t.Fatalf("handshake: %v", err)
    }

    enc := json.NewEncoder(conn)
    dec := json.NewDecoder(conn)

    // Send a request to the failing handler.
    if err := enc.Encode(map[string]interface{}{
        "type":    "request",
        "id":      "req-1",
        "command": "test.fail",
        "args":    map[string]interface{}{},
    }); err != nil {
        t.Fatalf("send request: %v", err)
    }

    var resp struct {
        Type  string `json:"type"`
        ID    string `json:"id"`
        OK    bool   `json:"ok"`
        Error struct {
            Code    string `json:"code"`
            Message string `json:"message"`
        } `json:"error"`
    }
    if err := dec.Decode(&resp); err != nil {
        t.Fatalf("decode response: %v", err)
    }

    if resp.OK {
        t.Error("VP-071 violated: ok=true after handler error")
    }
    if resp.Error.Code != "E-RPC-011" {
        t.Errorf("VP-071 violated: expected E-RPC-011, got %q", resp.Error.Code)
    }
    if resp.Error.Message != handlerErrMsg {
        t.Errorf("VP-071 violated: error message not verbatim: got %q, want %q",
            resp.Error.Message, handlerErrMsg)
    }
    if resp.ID != "req-1" {
        t.Errorf("VP-071 violated: response id mismatch: got %q", resp.ID)
    }

    // Connection must still be open — verify with a succeeding request.
    if err := enc.Encode(map[string]interface{}{
        "type":    "request",
        "id":      "req-2",
        "command": "test.ok",
        "args":    map[string]interface{}{},
    }); err != nil {
        t.Fatalf("VP-071 violated: connection closed after handler error: %v", err)
    }
    var resp2 struct{ OK bool `json:"ok"` }
    if err := dec.Decode(&resp2); err != nil {
        t.Fatalf("VP-071 violated: decode second response: %v", err)
    }
    if !resp2.OK {
        t.Error("VP-071 violated: second request after handler error returned ok=false")
    }
}
```
