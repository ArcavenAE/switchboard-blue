# VP-043: XOR FEC — Single Loss in Group Recoverable

## Property Statement

∀ group of N=4 data frames {f0, f1, f2, f3} and parity frame p = f0 ⊕ f1 ⊕ f2 ⊕ f3:
  ∀ i ∈ {0,1,2,3}:
    Recover({f0,..,f3} \ {fi}, p) == fi

When exactly one data frame is missing from a FEC group, the XOR parity
frame must allow recovery of the missing frame. This is the invariant for
N=4 XOR FEC as defined in BC-2.02.007.

## Source Contract

- **BC:** BC-2.02.007 — ARQ XOR FEC Single-Loss Recovery
- **Postcondition/Invariant:** Postcondition — "FEC Recover reconstructs missing frame from XOR parity."

## Proof Harness Skeleton

```go
package arq_test

import (
	"testing"

	"github.com/leanovate/gopter"
	"github.com/leanovate/gopter/gen"
	"github.com/leanovate/gopter/prop"

	"github.com/arcavenae/switchboard/internal/arq"
	"github.com/arcavenae/switchboard/internal/frame"
)

func TestProp_XOR_FEC_SingleLossRecoverable(t *testing.T) {
	properties := gopter.NewProperties(nil)

	properties.Property("any single missing frame recoverable from XOR parity", prop.ForAll(
		func(p0, p1, p2, p3 []byte) bool {
			// Pad all payloads to the same length.
			maxLen := len(p0)
			for _, p := range [][]byte{p1, p2, p3} {
				if len(p) > maxLen {
					maxLen = len(p)
				}
			}
			pad := func(b []byte) []byte {
				out := make([]byte, maxLen)
				copy(out, b)
				return out
			}
			payloads := [][]byte{pad(p0), pad(p1), pad(p2), pad(p3)}

			// Build frames.
			frames := make([]frame.Frame, 4)
			for i, pl := range payloads {
				frames[i].Sequence = uint64(i)
				frames[i].Payload = pl
			}

			// Build FEC group.
			group := arq.NewFECGroup(frames)

			// Drop each frame in turn and verify recovery.
			for dropIdx := 0; dropIdx < 4; dropIdx++ {
				remaining := make([]frame.Frame, 0, 3)
				for i, f := range frames {
					if i != dropIdx {
						remaining = append(remaining, f)
					}
				}
				recovered, err := group.Recover(remaining)
				if err != nil {
					return false
				}
				// Compare recovered payload to original.
				if len(recovered.Payload) != len(payloads[dropIdx]) {
					return false
				}
				for j := range recovered.Payload {
					if recovered.Payload[j] != payloads[dropIdx][j] {
						return false
					}
				}
			}
			return true
		},
		gen.SliceOfN(16, gen.UInt8()),
		gen.SliceOfN(16, gen.UInt8()),
		gen.SliceOfN(16, gen.UInt8()),
		gen.SliceOfN(16, gen.UInt8()),
	))

	properties.TestingRun(t)
}
```
