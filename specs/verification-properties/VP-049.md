# VP-049: sbctl Unified CLI with OpenSSH Auth

## Property Statement

sbctl with a valid --key, targeting each daemon type:
  router → successful response
  ∧ access → successful response
  ∧ console → successful response
  ∧ control → successful response

A single sbctl binary must authenticate successfully against all four daemon
types using the same OpenSSH-style key mechanism. No daemon-specific client
binary is required.

## Source Contract

- **BC:** BC-2.07.002 — sbctl unified CLI for all four daemon types with OpenSSH key authentication
- **Postcondition/Invariant:** Postcondition — "sbctl with valid key returns a valid response from router, access, console, and control daemons."

## Proof Method

| Method | Tool | Bounded? | Coverage |
|--------|------|----------|----------|
| e2e | go test (testing.T) | yes | full four-daemon stack; sbctl status against each; assert exit 0 and non-empty response |

## Proof Harness Skeleton

```go
//go:build integration

package mgmt_test

import (
	"context"
	"crypto/ed25519"
	"crypto/rand"
	"encoding/json"
	"net"
	"testing"
	"time"

	"github.com/arcavenae/switchboard/internal/mgmt"
)

// TestE2E_MgmtPlane_UnifiedAuth_VP049 verifies that a single operator key
// authenticates successfully against all four daemon types by starting four
// distinct mgmt.Server instances (one per mode) with mode-specific handler
// tables, as required by ARCH-12 §Story 3 (S-W5.02).
func TestE2E_MgmtPlane_UnifiedAuth_VP049(t *testing.T) {
	t.Parallel()

	ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
	t.Cleanup(cancel)

	// Operator keypair (distinct from daemon keys — primary case per PO ruling Q5).
	operatorPub, operatorPriv, err := ed25519.GenerateKey(rand.Reader)
	if err != nil {
		t.Fatalf("generate operator keypair: %v", err)
	}
	operatorKeySet := mgmt.NewOperatorKeySet([]ed25519.PublicKey{operatorPub})

	modes := []struct {
		name     string
		handlers []mgmt.Handler
	}{
		{"router", routerHandlers()},
		{"access", accessHandlers()},
		{"console", consoleHandlers()},
		{"control", controlHandlers()},
	}

	for _, mode := range modes {
		mode := mode
		t.Run(mode.name, func(t *testing.T) {
			t.Parallel()

			// Each mode gets its own ephemeral daemon keypair.
			daemonPub, daemonPriv, err := ed25519.GenerateKey(rand.Reader)
			if err != nil {
				t.Fatalf("%s: generate daemon keypair: %v", mode.name, err)
			}
			_ = daemonPub

			ln, err := net.Listen("tcp", "127.0.0.1:0")
			if err != nil {
				t.Fatalf("%s: listen: %v", mode.name, err)
			}
			t.Cleanup(func() { ln.Close() })

			srv, err := mgmt.NewServer(
				daemonPriv,
				operatorKeySet,
				mode.handlers,
				mgmt.WithHandshakeTimeout(500*time.Millisecond),
			)
			if err != nil {
				t.Fatalf("%s: NewServer: %v", mode.name, err)
			}

			serveCtx, serveCancel := context.WithCancel(ctx)
			t.Cleanup(serveCancel)
			go func() {
				if err := srv.Serve(serveCtx, ln); err != nil {
					// Serve returns nil on clean shutdown; any error here is unexpected.
					t.Errorf("%s: srv.Serve: %v", mode.name, err)
				}
			}()

			conn, err := net.Dial("tcp", ln.Addr().String())
			if err != nil {
				t.Fatalf("%s: dial: %v", mode.name, err)
			}
			t.Cleanup(func() { conn.Close() })

			authCtx, authCancel := context.WithTimeout(ctx, 5*time.Second)
			t.Cleanup(authCancel)
			if err := mgmt.Authenticate(authCtx, conn, operatorPriv); err != nil {
				t.Fatalf("%s: Authenticate: %v", mode.name, err)
			}

			// Send a "status" RPC with a non-constant request ID.
			reqID := nonConstantID()
			req := map[string]any{
				"type":    "request",
				"id":      reqID,
				"command": "status",
				"params":  map[string]any{},
			}
			if err := json.NewEncoder(conn).Encode(req); err != nil {
				t.Fatalf("%s: encode RPC request: %v", mode.name, err)
			}

			var resp struct {
				Type string `json:"type"`
				ID   string `json:"id"`
				OK   bool   `json:"ok"`
			}
			if err := json.NewDecoder(conn).Decode(&resp); err != nil {
				t.Fatalf("%s: decode RPC response: %v", mode.name, err)
			}
			if resp.Type != "response" {
				t.Errorf("%s: resp.Type = %q, want \"response\"", mode.name, resp.Type)
			}
			if resp.ID != reqID {
				t.Errorf("%s: resp.ID = %q, want %q (echo check)", mode.name, resp.ID, reqID)
			}
			if !resp.OK {
				t.Errorf("%s: RPC ok=false, expected success", mode.name)
			}
		})
	}
}

// TestE2E_MgmtPlane_BootstrapAuth_VP049 is the secondary variant: daemon key
// == operator key (bootstrap mode). Same wire-protocol assertions apply.
func TestE2E_MgmtPlane_BootstrapAuth_VP049(t *testing.T) {
	t.Parallel()

	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
	t.Cleanup(cancel)

	// Bootstrap: operator key IS the daemon key.
	daemonPub, daemonPriv, err := ed25519.GenerateKey(rand.Reader)
	if err != nil {
		t.Fatalf("generate daemon keypair: %v", err)
	}
	operatorKeySet := mgmt.NewOperatorKeySet([]ed25519.PublicKey{daemonPub})

	ln, err := net.Listen("tcp", "127.0.0.1:0")
	if err != nil {
		t.Fatalf("listen: %v", err)
	}
	t.Cleanup(func() { ln.Close() })

	srv, err := mgmt.NewServer(
		daemonPriv,
		operatorKeySet,
		routerHandlers(),
		mgmt.WithHandshakeTimeout(500*time.Millisecond),
	)
	if err != nil {
		t.Fatalf("NewServer: %v", err)
	}

	serveCtx, serveCancel := context.WithCancel(ctx)
	t.Cleanup(serveCancel)
	go srv.Serve(serveCtx, ln) //nolint:errcheck // test goroutine

	conn, err := net.Dial("tcp", ln.Addr().String())
	if err != nil {
		t.Fatalf("dial: %v", err)
	}
	t.Cleanup(func() { conn.Close() })

	authCtx, authCancel := context.WithTimeout(ctx, 5*time.Second)
	t.Cleanup(authCancel)
	if err := mgmt.Authenticate(authCtx, conn, daemonPriv); err != nil {
		t.Fatalf("bootstrap Authenticate: %v", err)
	}
}

// nonConstantID returns a non-repeating request ID (hex of nanosecond clock).
func nonConstantID() string {
	return fmt.Sprintf("%x", time.Now().UnixNano())
}
```

## Feasibility

VP-049 verifies the mgmt.Server auth handshake and RPC contract against four mgmt.Server
instances configured with distinct per-mode handler tables. The four daemon binary entrypoints
(`runRouter`, `runAccess`, `runConsole`, `runControl`) share the same mgmt.Server implementation
by ARCH-12 design; per-daemon binary wiring verification (goroutine lifecycle, config parsing,
signal handling) is deferred to a Wave-6 follow-up VP (placeholder VP-VW6.NN) once
`runRouter` and `runConsole` are unblocked from their `"not implemented"` stub state.
Bootstrap-mode auth (daemon key == operator key) retained as a secondary variant test
`TestE2E_MgmtPlane_BootstrapAuth_VP049` alongside the primary distinct-operator-key case.

## Feasibility Assessment

| Factor | Assessment | Notes |
|--------|-----------|-------|
| Input space size | four scenarios | one per daemon type (router, access, console, control) |
| Proof complexity | medium | four in-process mgmt.Server instances; no subprocess spawn |
| Tool support | go test (testing.T) + internal/mgmt | no external binary required |
| Estimated proof time | < 10s | four parallel subtests with 5s auth timeout each |
