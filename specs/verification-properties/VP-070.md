# VP-070: Unregistered RPC Command Returns E-RPC-010 In-Band; Connection Not Closed

## Property Statement

When an authenticated client sends an RPC request naming a command that is NOT
registered in the handler slice, `mgmt.Server` MUST:

1. Send an in-band error response:
   ```json
   {"type":"response","id":"<id>","ok":false,"error":{"code":"E-RPC-010","message":"unknown command: <cmd>"},"data":null}
   ```
   where `<id>` matches the request's `id` field and `<cmd>` is the unrecognized
   command name verbatim.
2. NOT close the connection. The client MUST be able to send further RPC requests
   on the same connection after receiving this response.
3. NOT use error code E-RPC-002. Any occurrence of `E-RPC-002` in `internal/mgmt`
   is a defect.

This is server-side only. It is distinct from the client-side `E-RPC-001` emitted
by `cmd/sbctl`.

## Source Contract

- **BC:** BC-2.07.004 PC-11 (Ruling C) — "unregistered RPC command → in-band
  response ok:false code E-RPC-010 'unknown command: <cmd>', connection NOT closed."

## Proof Method

| Method | Tool | Bounded? | Coverage |
|--------|------|----------|----------|

## Proof Harness Skeleton

```go
package mgmt_test

import (
    "context"
    "crypto/ed25519"
    "crypto/rand"
    "encoding/json"
    "net"
    "testing"
    "time"

    "github.com/arcavenae/switchboard/internal/mgmt"
)

// TestUnknownCommand_InBandError_VP070 verifies VP-070:
// Unknown RPC command → E-RPC-010 in-band; connection remains open.
func TestUnknownCommand_InBandError_VP070(t *testing.T) {
    t.Parallel()

    // Start a real server with no registered handlers.
    ln, err := net.Listen("tcp", "127.0.0.1:0")
    if err != nil {
        t.Fatalf("listen: %v", err)
    }
    t.Cleanup(func() { ln.Close() })

    _, daemonKey, _ := ed25519.GenerateKey(rand.Reader)
    opPub, opPriv, _ := ed25519.GenerateKey(rand.Reader)
    keySet := mgmt.NewOperatorKeySet([]ed25519.PublicKey{opPub})
    srv := mgmt.NewServer(ln, daemonKey, keySet, nil, "dev")

    ctx, cancel := context.WithCancel(context.Background())
    t.Cleanup(cancel)
    go srv.Serve(ctx) //nolint:errcheck

    // Connect and authenticate.
    conn, err := net.DialTimeout("tcp", ln.Addr().String(), time.Second)
    if err != nil {
        t.Fatalf("dial: %v", err)
    }
    t.Cleanup(func() { conn.Close() })

    // Perform handshake (read CHALLENGE, send CHALLENGE_RESPONSE).
    if err := doHandshake(t, conn, opPriv); err != nil {
        t.Fatalf("handshake: %v", err)
    }

    enc := json.NewEncoder(conn)
    dec := json.NewDecoder(conn)

    // Send an unknown command.
    if err := enc.Encode(map[string]interface{}{
        "type":    "request",
        "id":      "req-1",
        "command": "nonexistent.command",
        "args":    map[string]interface{}{},
    }); err != nil {
        t.Fatalf("send request: %v", err)
    }

    // Decode the response.
    var resp struct {
        Type  string `json:"type"`
        ID    string `json:"id"`
        OK    bool   `json:"ok"`
        Error struct {
            Code    string `json:"code"`
            Message string `json:"message"`
        } `json:"error"`
    }
    if err := dec.Decode(&resp); err != nil {
        t.Fatalf("decode response: %v", err)
    }

    if resp.OK {
        t.Error("VP-070 violated: ok=true for unknown command")
    }
    if resp.Error.Code != "E-RPC-010" {
        t.Errorf("VP-070 violated: expected E-RPC-010, got %q", resp.Error.Code)
    }
    if resp.ID != "req-1" {
        t.Errorf("VP-070 violated: response id mismatch: got %q", resp.ID)
    }

    // Connection must still be open — send another request.
    if err := enc.Encode(map[string]interface{}{
        "type":    "request",
        "id":      "req-2",
        "command": "another.unknown",
        "args":    map[string]interface{}{},
    }); err != nil {
        t.Fatalf("VP-070 violated: connection closed after first unknown command: %v", err)
    }
    var resp2 struct {
        OK    bool   `json:"ok"`
        Error struct{ Code string `json:"code"` } `json:"error"`
    }
    if err := dec.Decode(&resp2); err != nil {
        t.Fatalf("VP-070 violated: decode second response: %v", err)
    }
    if resp2.OK || resp2.Error.Code != "E-RPC-010" {
        t.Errorf("VP-070 violated: second unknown command did not return E-RPC-010")
    }
}
```
