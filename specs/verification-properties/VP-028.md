# VP-028: Config.Validate Rejects Out-of-Range tick_interval

## Property Statement

∀ tick_interval d:
  d < 5ms ∨ d > 50ms
  ⟹ Validate(config) returns a non-empty error slice
      ∧ at least one error references field 'tick_interval_upstream' or 'tick_interval_downstream'

A tick interval outside the allowed range [5ms, 50ms] is a configuration
error that must be surfaced explicitly by field name, not silently clamped
or ignored.

## Source Contract

- **BC:** BC-2.09.003 — Config.Validate Enforces tick_interval Range [5ms, 50ms]
- **Postcondition/Invariant:** Postcondition — "Out-of-range tick_interval produces a named field error."

## Proof Method

| Method | Tool | Bounded? | Coverage |
|--------|------|----------|----------|
| proptest | gopter v0.2.9+ | no | random durations below 5ms and above 50ms; asserts Validate returns non-empty errors with relevant field name |

## Proof Harness Skeleton

```go
package config_test

import (
	"testing"
	"time"

	"github.com/leanovate/gopter"
	"github.com/leanovate/gopter/gen"
	"github.com/leanovate/gopter/prop"

	"github.com/arcavenae/switchboard/internal/config"
)

func validBaseConfig() config.RouterConfig {
	return config.RouterConfig{
		TickIntervalUpstream:   10 * time.Millisecond,
		TickIntervalDownstream: 50 * time.Millisecond,
		// other required fields omitted; populated by test helper
	}
}

func TestProp_Config_RejectsOutOfRangeTickInterval(t *testing.T) {
	properties := gopter.NewProperties(nil)

	// Below minimum (< 5ms).
	properties.Property("tick_interval < 5ms produces field error", prop.ForAll(
		func(ms uint8) bool {
			if ms == 0 {
				ms = 1
			}
			d := time.Duration(ms%5) * time.Millisecond // 0–4ms
			if d == 0 {
				d = time.Millisecond
			}
			cfg := validBaseConfig()
			cfg.TickIntervalUpstream = d
			errs := config.Validate(cfg)
			if len(errs) == 0 {
				return false
			}
			for _, e := range errs {
				if e.Field == "tick_interval_upstream" || e.Field == "tick_interval_downstream" {
					return true
				}
			}
			return false
		},
		gen.UInt8Range(1, 4),
	))

	// Above maximum (> 50ms).
	properties.Property("tick_interval > 50ms produces field error", prop.ForAll(
		func(extraMs uint8) bool {
			d := 51*time.Millisecond + time.Duration(extraMs)*time.Millisecond
			cfg := validBaseConfig()
			cfg.TickIntervalUpstream = d
			errs := config.Validate(cfg)
			if len(errs) == 0 {
				return false
			}
			for _, e := range errs {
				if e.Field == "tick_interval_upstream" || e.Field == "tick_interval_downstream" {
					return true
				}
			}
			return false
		},
		gen.UInt8(),
	))

	properties.TestingRun(t)
}
```
