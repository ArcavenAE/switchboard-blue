# VP-050: Console Remotely Controllable via sbctl

## Story Trace

| Story | Version | Role |
|-------|---------|------|
| S-7.03 | v1.3 | implementing story — sbctl console attach/detach/switch via mgmt-plane Unix socket (ADR-006/ADR-012); BC-2.08.001 PC-1/PC-2/PC-3 |

## Property Statement

`sbctl console attach/detach/switch` commands sent to a running console daemon:
  are executed
  ∧ the console state changes accordingly

The sbctl CLI must be able to drive all three console lifecycle commands
remotely. Each command must produce an observable state change in the
console daemon.

## Source Contract

- **BC:** BC-2.08.001 — Console remotely controllable via sbctl: attach, detach, switch session, navigate
- **Postcondition/Invariant:** Postcondition — "attach/detach/switch commands execute and produce observable console state changes."

## Proof Method

| Method | Tool | Bounded? | Coverage |
|--------|------|----------|----------|
| e2e | go test (testing.T) | yes | running console daemon; sbctl attach → console attached; sbctl detach → console detached; sbctl switch → session changed |

## Proof Harness Skeleton

```go
//go:build integration

package sbctl_test

import (
	"bytes"
	"context"
	"os/exec"
	"testing"
	"time"

	"crypto/ed25519"
	"crypto/rand"
	"net"

	"github.com/arcavenae/switchboard/internal/mgmt"
	"github.com/arcavenae/switchboard/internal/session"
)

// TestE2E_Sbctl_Console_RemoteControl verifies that sbctl console
// attach/detach/switch commands change the running console daemon's state.
//
// NOTE: Uses in-process mgmt.NewServer + net.Listen pattern (per S-W5.02
// adversary Pass-1 Q4; no testenv package exists in this codebase).
// Real sbctl binary invoked via exec.CommandContext.
func TestE2E_Sbctl_Console_RemoteControl(t *testing.T) {
	t.Parallel()

	ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
	t.Cleanup(cancel)

	// Generate operator key pair for auth.
	pubKey, privKey, err := ed25519.GenerateKey(rand.Reader)
	if err != nil {
		t.Fatalf("generate key: %v", err)
	}
	_ = pubKey

	// Start in-process console daemon mgmt server (Unix socket).
	ln, err := net.Listen("unix", "")
	if err != nil {
		t.Fatalf("listen: %v", err)
	}
	t.Cleanup(func() { ln.Close() })
	consoleAddr := ln.Addr().String()

	// Build session store for the console daemon RPC handlers.
	sessStore := session.NewStore()
	sessionA := "session-alpha"
	sessionB := "session-beta"
	sessStore.Register(sessionA)
	sessStore.Register(sessionB)

	// Wire console RPC handlers: attach/detach/switch.
	handlers := buildConsoleHandlers(sessStore)

	daemonKey := make([]byte, ed25519.PrivateKeySize)
	copy(daemonKey, privKey)
	srv := mgmt.NewServer(daemonKey, "test-version", ln, nil, handlers)
	go srv.Serve(ctx)
	t.Cleanup(func() { srv.Shutdown() })

	// Write operator key to temp file for sbctl --key flag.
	keyFile := t.TempDir() + "/op.key"
	writeKeyFile(t, keyFile, privKey)

	sbctl := func(t *testing.T, args ...string) (stdout, stderr string) {
		t.Helper()
		base := []string{"--key", keyFile, "--target", consoleAddr}
		cmd := exec.CommandContext(ctx, "sbctl", append(base, args...)...)
		var out, errOut bytes.Buffer
		cmd.Stdout = &out
		cmd.Stderr = &errOut
		if err := cmd.Run(); err != nil {
			t.Fatalf("sbctl %v failed: %v\nstderr: %s", args, err, errOut.String())
		}
		return out.String(), errOut.String()
	}

	// --- attach ---
	sbctl(t, "console", "attach", "--session", sessionA)
	time.Sleep(200 * time.Millisecond)

	state := env.ConsoleState(t)
	if state.AttachedSession != sessionA {
		t.Errorf("after attach: expected session %s, got %s", sessionA, state.AttachedSession)
	}
	if !state.IsAttached {
		t.Error("after attach: console reports not attached")
	}

	// --- switch ---
	sbctl(t, "console", "switch", "--session", sessionB)
	time.Sleep(200 * time.Millisecond)

	state = env.ConsoleState(t)
	if state.AttachedSession != sessionB {
		t.Errorf("after switch: expected session %s, got %s", sessionB, state.AttachedSession)
	}

	// --- detach ---
	sbctl(t, "console", "detach")
	time.Sleep(200 * time.Millisecond)

	state = env.ConsoleState(t)
	if state.IsAttached {
		t.Error("after detach: console still reports attached")
	}
}
```
