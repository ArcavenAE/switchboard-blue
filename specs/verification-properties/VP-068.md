# VP-068: mgmt.NewServer Panics at Construction if len(daemonKey) != ed25519.PrivateKeySize

## Property Statement

`mgmt.NewServer(ln, daemonKey, operatorKeySet, handlers, daemonVersion)` MUST panic
at call time — before returning any value — whenever `len(daemonKey) != ed25519.PrivateKeySize`
(64 bytes). Specifically:

1. A nil `daemonKey` (len == 0) triggers the panic.
2. A non-nil but truncated key (0 < len < 64) triggers the panic.
3. A key of exactly 64 bytes does NOT trigger the panic.
4. The panic message MUST contain enough context for a developer to identify the
   cause (e.g., "mgmt.NewServer: daemonKey must be ed25519.PrivateKeySize bytes").

This fail-fast guard ensures that a nil or short key can never reach a live connection
goroutine where it would cause a nil-pointer dereference — a remote-panic DoS vector
(ARCH-12 Ruling A.2). It mirrors the existing `daemonVersion` emptiness check in
`NewServer`.

## Proof Harness Skeleton

```go
package mgmt_test

import (
    "crypto/ed25519"
    "crypto/rand"
    "net"
    "testing"

    "github.com/arcavenae/switchboard/internal/mgmt"
)

// TestNewServer_DaemonKeyPanic_VP068 verifies VP-068:
// NewServer panics if len(daemonKey) != ed25519.PrivateKeySize.
func TestNewServer_DaemonKeyPanic_VP068(t *testing.T) {
    t.Parallel()

    // Helper: a no-op listener we can pass to NewServer.
    ln, err := net.Listen("tcp", "127.0.0.1:0")
    if err != nil {
        t.Fatalf("listen: %v", err)
    }
    t.Cleanup(func() { ln.Close() })

    validKey, _, _ := ed25519.GenerateKey(rand.Reader)
    _ = validKey

    mustPanic := func(t *testing.T, name string, key ed25519.PrivateKey) {
        t.Helper()
        defer func() {
            if r := recover(); r == nil {
                t.Errorf("VP-068 violated (%s): NewServer did not panic with key len=%d", name, len(key))
            }
        }()
        // Attempt construction — must panic before returning.
        mgmt.NewServer(ln, key, nil, nil, "dev")
    }

    mustNotPanic := func(t *testing.T, name string, key ed25519.PrivateKey) {
        t.Helper()
        defer func() {
            if r := recover(); r != nil {
                t.Errorf("VP-068 violated (%s): NewServer panicked with valid key len=%d: %v", name, len(key), r)
            }
        }()
        _ = mgmt.NewServer(ln, key, nil, nil, "dev")
    }

    t.Run("nil_key_panics", func(t *testing.T) {
        t.Parallel()
        mustPanic(t, "nil", nil)
    })
    t.Run("empty_key_panics", func(t *testing.T) {
        t.Parallel()
        mustPanic(t, "empty", ed25519.PrivateKey{})
    })
    t.Run("short_key_32_panics", func(t *testing.T) {
        t.Parallel()
        mustPanic(t, "len=32", make(ed25519.PrivateKey, 32))
    })
    t.Run("short_key_63_panics", func(t *testing.T) {
        t.Parallel()
        mustPanic(t, "len=63", make(ed25519.PrivateKey, 63))
    })
    t.Run("valid_key_64_no_panic", func(t *testing.T) {
        t.Parallel()
        _, key, _ := ed25519.GenerateKey(rand.Reader)
        mustNotPanic(t, "len=64", key)
    })
}
```
