# VP-048: Control Node Creates/Destroys SVTNs

## Property Statement

(1) svtn create → the new SVTN appears in svtn list
(2) svtn destroy → the SVTN no longer appears in svtn list
    ∧ nodes can no longer admit to the destroyed SVTN

SVTN lifecycle management must be immediately consistent: a create must be
visible in the next list call; a destroy must prevent any further admission.

## Source Contract

- **BC:** BC-2.07.001 — Control Node SVTN Create/Destroy Operations
- **Postcondition/Invariant:** Postcondition — "(1) created SVTN appears in list; (2) destroyed SVTN absent from list and admission rejected."

## Proof Harness Skeleton

```go
//go:build integration

package svtnmgmt_test

import (
	"context"
	"testing"
	"time"

	"github.com/arcavenae/switchboard/internal/testenv"
)

// TestIntegration_ControlNode_SVTNCreateDestroy verifies create and destroy
// lifecycle operations on the control node.
func TestIntegration_ControlNode_SVTNCreateDestroy(t *testing.T) {
	t.Parallel()

	ctx, cancel := context.WithTimeout(context.Background(), 20*time.Second)
	t.Cleanup(cancel)

	env := testenv.New(t, ctx)
	t.Cleanup(env.Close)

	// (1) Create a new SVTN; it must appear in svtn list.
	t.Run("create appears in list", func(t *testing.T) {
		svtnID := env.ControlNode(t).CreateSVTN(t, "test-svtn-a")
		svtns, err := env.ControlNode(t).ListSVTNs(t)
		if err != nil {
			t.Fatalf("ListSVTNs error: %v", err)
		}
		found := false
		for _, s := range svtns {
			if s.ID == svtnID {
				found = true
				break
			}
		}
		if !found {
			t.Errorf("newly created SVTN %v not found in list", svtnID)
		}
	})

	// (2) Destroy a SVTN; it must not appear in list and admission must fail.
	t.Run("destroy absent from list and blocks admission", func(t *testing.T) {
		svtnID := env.ControlNode(t).CreateSVTN(t, "test-svtn-b")
		env.ControlNode(t).DestroySVTN(t, svtnID)

		// Give control plane a moment to propagate.
		time.Sleep(100 * time.Millisecond)

		svtns, err := env.ControlNode(t).ListSVTNs(t)
		if err != nil {
			t.Fatalf("ListSVTNs error: %v", err)
		}
		for _, s := range svtns {
			if s.ID == svtnID {
				t.Errorf("destroyed SVTN %v still appears in list", svtnID)
			}
		}

		// Attempt admission to the destroyed SVTN.
		creds := env.GenerateCredentialsForSVTN(t, svtnID)
		err = env.ConnectWithKey(t, creds)
		if err == nil {
			t.Error("expected admission to destroyed SVTN to fail, but it succeeded")
		}
	})
}
```
