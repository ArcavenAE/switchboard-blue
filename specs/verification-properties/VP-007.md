# VP-007: Admission Private Key Bytes Never Appear in ChallengeResponse Wire Struct

## Property Statement

∀ privKey [64]byte (ed25519 private key), challenge [32]byte:
  let resp = SignChallenge(privKey, challenge)
  let wire = MarshalChallengeResponse(resp)
  ¬ (∃ i : wire[i:i+32] == privKey[0:32] ∨ wire[i:i+32] == privKey[32:64])

Neither the seed (first 32 bytes) nor the public-key portion (last 32 bytes)
of the ed25519 private key appear as a contiguous 32-byte substring anywhere
in the serialized ChallengeResponse. The response contains only the 64-byte
signature and the public key; the private key material must never transit.

## Source Contract

- **BC:** BC-2.05.001 — Tier 1 SVTN admission via signed key challenge
- **BC:** BC-2.05.007 — Node private keys never transit the network under any condition
- **Postcondition/Invariant:** BC-2.05.007 Postcondition 1 — "The private key bytes are not present in any outgoing network frame." BC-2.05.001 Invariant DI-002 — "The node's private key is used only to sign the challenge locally. It never leaves the node."

## Proof Harness Skeleton

```go
package admission_test

import (
	"bytes"
	"crypto/ed25519"
	"crypto/rand"
	"testing"

	"github.com/leanovate/gopter"
	"github.com/leanovate/gopter/gen"
	"github.com/leanovate/gopter/prop"

	"github.com/arcavenae/switchboard/internal/admission"
)

// containsSubslice reports whether haystack contains needle as a contiguous
// run of bytes.
func containsSubslice(haystack, needle []byte) bool {
	if len(needle) == 0 || len(needle) > len(haystack) {
		return false
	}
	for i := 0; i <= len(haystack)-len(needle); i++ {
		if bytes.Equal(haystack[i:i+len(needle)], needle) {
			return true
		}
	}
	return false
}

func TestPropPrivKeyNotInChallengeResponse(t *testing.T) {
	properties := gopter.NewProperties(nil)

	properties.Property("serialized ChallengeResponse contains no private key material", prop.ForAll(
		func(challengeBytes []byte) bool {
			// Generate a fresh ed25519 keypair for each test case.
			pubKey, privKey, err := ed25519.GenerateKey(rand.Reader)
			if err != nil {
				return true // key generation failure; skip rather than false-positive
			}
			_ = pubKey

			var challenge [32]byte
			copy(challenge[:], challengeBytes)

			resp, err := admission.SignChallenge(privKey, challenge)
			if err != nil {
				return true // signing error; skip
			}

			wire, err := admission.MarshalChallengeResponse(resp)
			if err != nil {
				return true
			}

			// Private key seed (first 32 bytes of ed25519 private key).
			seed := []byte(privKey)[:32]
			if containsSubslice(wire, seed) {
				return false
			}

			// The full private key is 64 bytes; check seed portion only —
			// last 32 bytes are the public key, which IS expected in the wire format.
			return true
		},
		gen.SliceOfN(32, gen.UInt8()),
	))

	properties.TestingRun(t)
}
```
