# VP-041: Tick Regularity — p99 Jitter ≤ 2ms

## Property Statement

Over 1,000 consecutive ticks:
  p99(|actual_interval - configured_interval|) ≤ 2ms

This covers NFR-009: tick jitter must not exceed 2ms at the 99th percentile.
High jitter causes perceptible keystroke latency variation and degrades
the interactive experience even when mean latency is within bounds.

## Source Contract

- **BC:** BC-2.01.001 — HalfChannel Tick Regularity (NFR-009)
- **Postcondition/Invariant:** NFR — "p99 tick jitter ≤ 2ms over 1,000 consecutive ticks."

## Proof Harness Skeleton

```go
package halfchannel_test

import (
	"sort"
	"testing"
	"time"

	"github.com/arcavenae/switchboard/internal/halfchannel"
)

// BenchmarkTick_Jitter measures p99 deviation from the configured tick
// interval over 1,000 consecutive ticks.
func BenchmarkTick_Jitter(b *testing.B) {
	const (
		configuredInterval = 10 * time.Millisecond
		tickCount          = 1000
		maxP99Jitter       = 2 * time.Millisecond
	)

	ch := halfchannel.New(halfchannel.State{})
	deviations := make([]time.Duration, 0, tickCount)

	b.ResetTimer()
	prev := time.Now()
	for i := 0; i < tickCount; i++ {
		// Block until the next scheduled tick.
		time.Sleep(configuredInterval - time.Since(prev))
		now := time.Now()
		actual := now.Sub(prev)
		prev = now

		_ = ch.Tick(nil)

		dev := actual - configuredInterval
		if dev < 0 {
			dev = -dev
		}
		deviations = append(deviations, dev)
	}
	b.StopTimer()

	sort.Slice(deviations, func(i, j int) bool {
		return deviations[i] < deviations[j]
	})

	p99idx := int(float64(len(deviations)) * 0.99)
	p99 := deviations[p99idx]

	b.ReportMetric(float64(p99.Milliseconds()), "p99_jitter_ms")

	if p99 > maxP99Jitter {
		b.Errorf("p99 jitter %v exceeds limit %v", p99, maxP99Jitter)
	}
}
```
