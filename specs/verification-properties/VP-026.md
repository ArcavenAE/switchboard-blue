# VP-026: PathScore Ranking Is Transitive

## Property Statement

∀ paths A, B, C:
  Score(A) < Score(B) ∧ Score(B) < Score(C)
  ⟹ Rank(A) < Rank(B) < Rank(C)

The path scoring function must induce a total order consistent with
transitivity. This ensures the multipath scheduler always selects the
globally optimal path and not a locally optimal one that is inconsistent
with the ranking of other paths.

## Source Contract

- **BC:** BC-2.02.003 — Per-path RTT and loss tracked via keep-alive probes; paths ranked by quality
- **Postcondition/Invariant:** Invariant — "Score ordering is transitive: Score(A)<Score(B) and Score(B)<Score(C) implies Score(A)<Score(C)."

| Method | Tool | Bounded? | Coverage |

## Proof Harness Skeleton

```go
package paths_test

import (
	"sort"
	"testing"

	"github.com/leanovate/gopter"
	"github.com/leanovate/gopter/gen"
	"github.com/leanovate/gopter/prop"

	"github.com/arcavenae/switchboard/internal/paths"
)

func genPathMetrics() gopter.Gen {
	return gopter.CombineGens(
		gen.UInt32Range(0, 1000),  // rtt_ms
		gen.Float64Range(0, 1.0),  // loss fraction
		gen.UInt32Range(0, 10000), // jitter_us
	).Map(func(vals []interface{}) paths.Metrics {
		return paths.Metrics{
			RTTMs:    vals[0].(uint32),
			LossFrac: vals[1].(float64),
			JitterUs: vals[2].(uint32),
		}
	})
}

func TestProp_PathScore_Transitive(t *testing.T) {
	properties := gopter.NewProperties(nil)

	properties.Property("Score ordering is transitive", prop.ForAll(
		func(a, b, c paths.Metrics) bool {
			sa := paths.Score(a)
			sb := paths.Score(b)
			sc := paths.Score(c)

			// Collect and sort; verify rank order matches score order.
			type scoredPath struct {
				score float64
				label string
			}
			scored := []scoredPath{
				{sa, "A"},
				{sb, "B"},
				{sc, "C"},
			}
			sort.Slice(scored, func(i, j int) bool {
				return scored[i].score < scored[j].score
			})

			// After sorting by score, ranked positions should be consistent.
			// Transitivity: if sa < sb and sb < sc then sa < sc.
			if sa < sb && sb < sc {
				return sa < sc
			}
			if sc < sb && sb < sa {
				return sc < sa
			}
			// Any other ordering is also acceptable (ties, partial order) —
			// we only assert the strict chain case above.
			return true
		},
		genPathMetrics(),
		genPathMetrics(),
		genPathMetrics(),
	))

	properties.TestingRun(t)
}
```
