# VP-029: Config.Validate Rejects Missing Required Fields

## Property Statement

∀ required field F in {RouterConfig, AccessConfig, ConsoleConfig, ControlConfig}:
  omitting F (zero value) causes Validate(config) to return an error

Each required field must be independently validated. Omitting any single
required field must produce at least one error; the error slice must not be
empty when any required field is absent.

## Source Contract

- **BC:** BC-2.09.003 — Config.Validate Enforces Required Fields
- **Postcondition/Invariant:** Postcondition — "Missing required field produces a non-empty error slice."

## Proof Method

| Method | Tool | Bounded? | Coverage |
|--------|------|----------|----------|
| proptest | gopter v0.2.9+ | no | for each config type, iterate over required fields; zero each one individually and assert Validate returns non-empty errors |

## Proof Harness Skeleton

```go
package config_test

import (
	"testing"

	"github.com/arcavenae/switchboard/internal/config"
)

// fullyValidRouterConfig returns a RouterConfig with all required fields set.
func fullyValidRouterConfig() config.RouterConfig {
	return config.RouterConfig{
		ListenAddr:             "127.0.0.1:7000",
		SVTNID:                 [16]byte{1},
		TickIntervalUpstream:   10_000_000,   // 10ms
		TickIntervalDownstream: 50_000_000,   // 50ms
	}
}

func fullyValidAccessConfig() config.AccessConfig {
	return config.AccessConfig{
		RouterAddr: "127.0.0.1:7000",
		SVTNID:     [16]byte{1},
	}
}

func fullyValidConsoleConfig() config.ConsoleConfig {
	return config.ConsoleConfig{
		AccessAddr: "127.0.0.1:7001",
	}
}

func fullyValidControlConfig() config.ControlConfig {
	return config.ControlConfig{
		ListenAddr: "127.0.0.1:7002",
	}
}

// TestConfig_ValidateRejectsZeroListenAddr tests a representative required
// field for each config type. The full property is confirmed by running this
// pattern for every required field.
func TestConfig_ValidateRejectsZeroListenAddr(t *testing.T) {
	cases := []struct {
		name   string
		mutateFn func() interface{}
		validateFn func(interface{}) []config.FieldError
	}{
		{
			name: "RouterConfig missing ListenAddr",
			mutateFn: func() interface{} {
				c := fullyValidRouterConfig()
				c.ListenAddr = ""
				return c
			},
			validateFn: func(v interface{}) []config.FieldError {
				return config.ValidateRouter(v.(config.RouterConfig))
			},
		},
		{
			name: "AccessConfig missing RouterAddr",
			mutateFn: func() interface{} {
				c := fullyValidAccessConfig()
				c.RouterAddr = ""
				return c
			},
			validateFn: func(v interface{}) []config.FieldError {
				return config.ValidateAccess(v.(config.AccessConfig))
			},
		},
		{
			name: "ConsoleConfig missing AccessAddr",
			mutateFn: func() interface{} {
				c := fullyValidConsoleConfig()
				c.AccessAddr = ""
				return c
			},
			validateFn: func(v interface{}) []config.FieldError {
				return config.ValidateConsole(v.(config.ConsoleConfig))
			},
		},
		{
			name: "ControlConfig missing ListenAddr",
			mutateFn: func() interface{} {
				c := fullyValidControlConfig()
				c.ListenAddr = ""
				return c
			},
			validateFn: func(v interface{}) []config.FieldError {
				return config.ValidateControl(v.(config.ControlConfig))
			},
		},
	}

	for _, tc := range cases {
		tc := tc
		t.Run(tc.name, func(t *testing.T) {
			t.Parallel()
			cfg := tc.mutateFn()
			errs := tc.validateFn(cfg)
			if len(errs) == 0 {
				t.Errorf("expected validation errors for %q, got none", tc.name)
			}
		})
	}
}
```
