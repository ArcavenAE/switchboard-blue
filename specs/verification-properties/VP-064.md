# VP-064: Management Server Rejects Unauthenticated Connections — AUTH_FAIL + Close, No RPC Dispatched

## Property Statement

∀ connection to `mgmt.Server` where the client either:
- (a) does not send a CHALLENGE_RESPONSE after receiving the CHALLENGE, or
- (b) sends a CHALLENGE_RESPONSE with a `pubkey` not in `OperatorKeySet`, or
- (c) sends a CHALLENGE_RESPONSE with a signature that does not verify against `pubkey`,

the server MUST:
1. Send an AUTH_FAIL message: `{"type":"auth_fail","code":"E-ADM-010","message":"authentication failed"}`
2. Close the connection immediately after sending AUTH_FAIL.
3. Never dispatch any RPC command to a registered handler on this connection.
4. Never emit any RPC response data (`"type":"response"`) on this connection.

The property is fail-closed: the default outcome for any non-passing handshake is
rejection. There is no code path that reaches RPC dispatch without a verified
AUTH_OK.

Additionally, AUTH_FAIL responses are indistinguishable between cases (b) and (c) —
no timing or content oracle differentiates "key not in set" from "wrong signature."

## Source Contract

- **BC:** BC-2.07.004 — Daemon Management Server Authenticates All Connections via
  Ed25519 Challenge-Response (Fail-Closed)
- **Postcondition PC-2:** unauthenticated connections rejected with E-ADM-010 + close.
- **Postcondition PC-4:** AUTH_FAIL closes connection without processing any RPC.
- **Postcondition PC-5:** all RPC commands require prior successful auth.
- **Postcondition PC-8:** constant-time comparison, no oracle.
- **Invariant 1:** fail-closed — default outcome is rejection.

## Proof Method

| Method | Tool | Bounded? | Coverage |
|--------|------|----------|----------|
| integration | `go test` + real Unix socket (temp dir) | yes | Three sub-cases: no response, wrong key, correct key wrong signature |

## Proof Harness Skeleton

```go
package mgmt_test

import (
    "context"
    "crypto/ed25519"
    "crypto/rand"
    "encoding/json"
    "net"
    "path/filepath"
    "testing"
    "time"

    "github.com/arcavenae/switchboard/internal/mgmt"
)

// TestMgmtServer_RejectsUnauthenticated_VP064 verifies VP-064:
// connections that fail the handshake receive AUTH_FAIL + close; no RPC dispatched.
func TestMgmtServer_RejectsUnauthenticated_VP064(t *testing.T) {
    t.Parallel()

    // Generate daemon keypair.
    daemonPub, daemonPriv, err := ed25519.GenerateKey(rand.Reader)
    if err != nil {
        t.Fatalf("generate daemon key: %v", err)
    }
    _ = daemonPub

    // Generate an authorized operator keypair.
    _, authorizedPriv, err := ed25519.GenerateKey(rand.Reader)
    if err != nil {
        t.Fatalf("generate authorized key: %v", err)
    }

    // Generate an unauthorized keypair (not in key set).
    unauthorizedPub, unauthorizedPriv, err := ed25519.GenerateKey(rand.Reader)
    if err != nil {
        t.Fatalf("generate unauthorized key: %v", err)
    }
    _ = unauthorizedPriv

    // Build operator key set with ONLY authorizedPriv's public key.
    authorizedKeySet := mgmt.NewOperatorKeySet([]ed25519.PublicKey{
        authorizedPriv.Public().(ed25519.PublicKey),
    })

    rpcCallCount := 0
    handlers := []mgmt.Handler{
        {
            Command: "test.echo",
            Fn: func(_ context.Context, _ json.RawMessage) (any, error) {
                rpcCallCount++ // must never increment in this test
                return map[string]string{"echo": "ok"}, nil
            },
        },
    }

    t.Run("wrong_key_not_in_set", func(t *testing.T) {
        t.Parallel()
        sockDir := t.TempDir()
        sockPath := filepath.Join(sockDir, "mgmt.sock")
        ln, err := net.Listen("unix", sockPath)
        if err != nil {
            t.Fatalf("listen: %v", err)
        }
        defer ln.Close()
        go mgmt.NewServer(ln, daemonPriv, authorizedKeySet, handlers, "dev").Serve(context.Background()) //nolint:errcheck
        client, err := net.Dial("unix", sockPath)
        if err != nil {
            t.Fatalf("dial: %v", err)
        }
        defer client.Close()

        // Read CHALLENGE.
        challenge := readMessage(t, client)
        if challenge["type"] != "challenge" {
            t.Fatalf("expected challenge, got: %v", challenge["type"])
        }
        nonceB64 := challenge["nonce"].(string)
        nonceBytes := mustDecodeBase64URL(t, nonceB64)

        // Sign with unauthorized key.
        sig := ed25519.Sign(unauthorizedPriv, nonceBytes)
        sendMessage(t, client, map[string]interface{}{
            "type":      "challenge_response",
            "nonce_sig": encodeBase64URL(sig),
            "pubkey":    encodeBase64URL([]byte(unauthorizedPub)),
        })

        // Expect AUTH_FAIL.
        resp := readMessage(t, client)
        if resp["type"] != "auth_fail" {
            t.Errorf("expected auth_fail, got: %v", resp["type"])
        }
        if resp["code"] != "E-ADM-010" {
            t.Errorf("expected E-ADM-010, got: %v", resp["code"])
        }

        // Connection should be closed now; no RPC response should arrive.
        client.SetReadDeadline(time.Now().Add(200 * time.Millisecond))
        extra := readMessage(t, client)
        if extra != nil {
            t.Errorf("expected closed connection, got extra message: %v", extra)
        }
    })

    t.Run("wrong_signature_recognized_key", func(t *testing.T) {
        t.Parallel()
        // Same test but uses the authorized public key with a wrong signature.
        sockDir := t.TempDir()
        sockPath := filepath.Join(sockDir, "mgmt.sock")
        ln, err := net.Listen("unix", sockPath)
        if err != nil {
            t.Fatalf("listen: %v", err)
        }
        defer ln.Close()
        go mgmt.NewServer(ln, daemonPriv, authorizedKeySet, handlers, "dev").Serve(context.Background()) //nolint:errcheck
        client, err := net.Dial("unix", sockPath)
        if err != nil {
            t.Fatalf("dial: %v", err)
        }
        defer client.Close()

        challenge := readMessage(t, client)
        nonceBytes := mustDecodeBase64URL(t, challenge["nonce"].(string))

        // Tamper with nonce before signing.
        tampered := append([]byte(nil), nonceBytes...)
        tampered[0] ^= 0xFF
        sig := ed25519.Sign(authorizedPriv, tampered)
        authorizedPub := authorizedPriv.Public().(ed25519.PublicKey)
        sendMessage(t, client, map[string]interface{}{
            "type":      "challenge_response",
            "nonce_sig": encodeBase64URL(sig),
            "pubkey":    encodeBase64URL([]byte(authorizedPub)),
        })

        resp := readMessage(t, client)
        if resp["type"] != "auth_fail" {
            t.Errorf("expected auth_fail, got: %v", resp["type"])
        }
        if resp["code"] != "E-ADM-010" {
            t.Errorf("expected E-ADM-010, got: %v", resp["code"])
        }
    })

    if rpcCallCount != 0 {
        t.Errorf("VP-064 violated: RPC handler called %d times on unauthenticated connections", rpcCallCount)
    }
}
```
